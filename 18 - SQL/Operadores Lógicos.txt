-- SELECT usando o operador AND
SELECT * FROM funcionarios WHERE id_funcionario >= 4 AND id_funcionario <= 7;

-- SELECT usando o operador OR
SELECT * FROM funcionarios WHERE id_funcionario >= 11 OR nome LIKE 'P%';

-- SELECT usando o operador BETWEEN
SELECT * FROM funcionarios WHERE id_funcionario BETWEEN 4 AND 8;

-- SELECT usando o operador NOT BETWEEN
SELECT * FROM funcionarios WHERE id_funcionario NOT BETWEEN 4 AND 7;

-- SELECT usando o operador IN
SELECT * FROM funcionarios WHERE id_departamento IN (1, 2, 5);

-- SELECT usando o operador NOT IN
SELECT * FROM funcionarios WHERE id_departamento NOT IN (2);

-- SELECT usando o operador IS e NULL
SELECT * FROM funcionarios WHERE id_departamento IS NULL;

-- SELECT usando o operador NOT NULL
SELECT * FROM funcionarios WHERE id_departamento IS NOT NULL;
